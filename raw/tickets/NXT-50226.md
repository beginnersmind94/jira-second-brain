# NXT-50226 — LMA - License Keys - Backend

## Description
We need to update the backend structure to accommodate the new method of handling licenses

*Notes:*

* This story is to update the backend to account for the new concepts
* Licenses already exists, but it is not meeting our needs
* We will create a new model based of the old model with the goal of leveraging the existing work, but adding in the missing functionality
* The biggest changes are:
** We are adding the concept of a License Key which acts as a container object that licenses will belong to
*** A License Key has many licenses
** The License Key will be managed by a new role called License Management role (LMA role)
** Districts can have many licenses, even for the same module
** To remove a license, we will generate a new one with a negative number
** Moving forward, to know if a district is licensed, we will sum all active licenses
*** Ex: 
**** First created 10 accountability licenses in a key
**** Then removed 3 licenses in a new key
**** Therefore, summing, they have 7 active licenses
** We will not be able to delete a license once it is created and linked to a district
** Operations like adding more, removing some, renewing or canceling will all generate new license keys with modified licenses counts
* We will run 2 license features in parallel until all are migrated over

*Requirements:*

* Update the backend to authorize modules and features based upon Licenses so that they account for the new structure
* Create a way to run License Management in either old way or new way per district and be able to turn them on or off

DB Structure



CREATE TABLE PR_SCLicense (
    SC_LicenseID INT IDENTITY(1,1) PRIMARY KEY,
    LicenseKey NVARCHAR(50) UNIQUE NOT NULL,
    RegionId INT FOREIGN KEY REFERENCES PR_Region(RegionId) ON DELETE CASCADE,
    PurchaseDate DATE NOT NULL,
    StartDate DATE NOT NULL,
    EndDate DATE NOT NULL,
    IsRenewal BIT DEFAULT 0,
    Description NVARCHAR(MAX),
    LicenseStatusId INT FOREIGN KEY REFERENCES PR_SCLicenseStatus(SC_LicenseStatusId),
    ParentLicenseID INT NULL FOREIGN KEY REFERENCES PR_SCLicense(SC_LicenseID),
    GeneratedBy NVARCHAR(255),
    GeneratedDate DATE DEFAULT GETDATE(),
    ActivityStatus NVARCHAR(50),
	CreatedBy INT NULL,
	UpdatedBy INT NULL,
    CreatedOn DATETIME DEFAULT GETDATE(),
    UpdatedOn DATETIME DEFAULT GETDATE()
);

CREATE TABLE PR_SCLicenseCancellation (
    SC_LicenseCancellationID INT IDENTITY(1,1) PRIMARY KEY,
    SC_LicenseID INT NULL FOREIGN KEY REFERENCES PR_SCLicense(SC_LicenseID),
	Comments VARCHAR(MAX) NULL,
	CancellationRequestDate DATETIME,
	CancellationEffectiveDate DATETIME,
	CancellationRequestedBy INT,
	CreatedBy INT NULL,
	CreatedOn DATETIME DEFAULT GETDATE(),
	UpdatedBy INT NULL,
	UpdatedOn DATETIME DEFAULT GETDATE()
);

CREATE TABLE PR_SCLicenseModules (
    SC_LicenseModuleID INT IDENTITY(1,1) PRIMARY KEY,
    LicenseID INT FOREIGN KEY REFERENCES PR_SCLicense(SC_LicenseID) ON DELETE CASCADE,
    ModuleID INT FOREIGN KEY REFERENCES NF_Module(ModuleID) ON DELETE CASCADE,
    LicenseName NVARCHAR(255) NOT NULL,
    LicenseTypeId INT FOREIGN KEY REFERENCES PR_SCLicenseTypes(LicenseTypeId) ON DELETE CASCADE,
    LicenseCount INT NULL,  -- 'All' can be represented by NULL or a specific number
    CreatedAt DATETIME DEFAULT GETDATE(),
    UpdatedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE [dbo].[PR_SCLicenseModuleSite] (
    [LicenseModuleSiteId] INT           IDENTITY (1, 1) NOT NULL,
    [SC_LicenseModuleId]     INT           NOT NULL,
    [SiteId]              SMALLINT      NOT NULL,
    [IsActive]            BIT           CONSTRAINT [DF_PR_SCLicenseModuleSite_IsActive] DEFAULT ((0)) NOT NULL,
    [IsDeleted]           BIT           CONSTRAINT [DF_PR_SCLicenseModuleSite_IsDeleted] DEFAULT ((0)) NOT NULL,
    [IsUsing]             BIT           CONSTRAINT [DF_PR_SCLicenseModuleSite_IsUsing] DEFAULT ((0)) NOT NULL,
    [CreatedOn]           DATETIME2 (7) NOT NULL,
    [CreatedBy]           INT           NOT NULL,
    [UpdatedOn]           DATETIME2 (7) NULL,
    [UpdatedBy]           INT           NULL,
    CONSTRAINT [PK_PR_SCLicenseModuleSiteId] PRIMARY KEY CLUSTERED ([LicenseModuleSiteId] ASC),
    CONSTRAINT [FK_PR_SCLicenseModuleSite_CreatedBy] FOREIGN KEY ([CreatedBy]) REFERENCES [dbo].[NF_User] ([UserId]),
    CONSTRAINT [FK_PR_SCLicenseModuleSite_SC_LicenseModuleID] FOREIGN KEY ([SC_LicenseModuleId]) REFERENCES [dbo].[PR_SCLicenseModules] ([SC_LicenseModuleId]),
    CONSTRAINT [FK_PR_SCLicenseModuleSite_SiteId] FOREIGN KEY ([SiteId]) REFERENCES [dbo].[PR_Site] ([SiteId]),
    CONSTRAINT [FK_PR_SCLicenseModuleSite_UpdatedBy] FOREIGN KEY ([UpdatedBy]) REFERENCES [dbo].[NF_User] ([UserId])
);

CREATE TABLE PR_SCLicenseTypes
(
	LicenseTypeId INT IDENTITY(1,1) PRIMARY KEY,
	LicenseTypeName NVARCHAR(50) NOT NULL,
	Description NVARCHAR(500) NULL,
	CreatedBy INT NULL,
	CreatedOn DATETIME null,
	UpdatedBy INT NULL,
	UpdatedOn DATETIME NULL
)

CREATE TABLE PR_SCLicenseStatus
(
	SC_LicenseStatusId INT IDENTITY(1,1) PRIMARY KEY,
	LicenseStatusName NVARCHAR(50) NOT NULL,
	CreatedBy INT NULL,
	CreatedOn DATETIME null,
	UpdatedBy INT NULL,
	UpdatedOn DATETIME NULL
)
