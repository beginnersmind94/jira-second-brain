# NXT-64910 — WS - Filters - Sites Multi-Select - PE

## Description
As a Workspace user, I want to be able to select sites more efficiently, so that I can be faster in filtering my data.

*Requirements*

* This story is for the filters on the main page
* It applies only to PrimeroEdge Districts (SC districts are in a related story)
* In the drop down list for the sites, do the following:
** Be able to multi-select

!image-20250806-205742.png|width=285,height=348,alt="image-20250806-205742.png"!

{noformat}declare
	@regionId int=3
	,@userId int=319
	,@primeroUserId uniqueidentifier=null


-- Get PrimeroUserId
SELECT 
    @primeroUserId = u.PrimeroUserId
FROM
    dbo.NF_User u with(nolock)
WHERE
    u.UserId = @userId;


declare @tblSite table
(
	PrimeroSchoolId uniqueidentifier
)

insert into @tblSite(PrimeroSchoolId)
SELECT 
	rs.SchoolId    
FROM    
    dbo.fx_Schools_ByUserIdAndRegionId(@primeroUserId, @RegionId, NULL, 'pos') as rs
    
--SiteType:
SELECT  distinct
    st.SiteTypeId
	,st.Description as SiteType
FROM    
    @tblSite rs
    inner join dbo.PR_Site s with(nolock) on s.PrimeroSchoolId = rs.PrimeroSchoolId
	inner join dbo.PR_SiteType st with(nolock) on st.SiteTypeId=s.SiteTypeId
WHERE
    s.DistrictRegionId = @regionId
    and s.IsActive = 1
	and st.RegionId=@regionId

order by
	st.Description


--Site:
SELECT
    s.SiteId
	,s.SiteName
FROM    
    @tblSite rs
    inner join dbo.PR_Site s with(nolock) on s.PrimeroSchoolId = rs.PrimeroSchoolId
WHERE
    s.DistrictRegionId = @regionId
    and s.IsActive = 1
order by
	s.SiteName{noformat}
